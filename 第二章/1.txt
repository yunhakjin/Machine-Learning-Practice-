#encoding:utf-8
from numpy import *
import matplotlib
import matplotlib.pyplot as plt

def entity(filename):
    file=open(filename)
    fileLine=file.readlines()
    classLabel=[] #目标分类
    index=0
    mat=zeros((len(fileLine),3))
    for line in fileLine:
        line=line.strip()#截取所有的回车符
        entity=line.split('\t')#使用\t作为分割元素的基本标志
        mat[index,:]=entity[0:3]
        classLabel.append(int(entity[-1]))
        index=index+1
    return mat,classLabel

mat,classLabel=entity('C:/Users/yxz/Desktop/data.txt')
fig=plt.figure()
ax=fig.add_subplot(111)
ax.scatter(mat[:,1],mat[:,2])
plt.show()
